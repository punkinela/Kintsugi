  const selfDoubt = ['maybe', 'perhaps', 'might', 'possibly'];
  const regret = ['should have', 'could have', 'wish I', 'if only'];

  // Pattern matching with context awareness
  for (let pattern of [...minimizingWords, ...selfDoubt, ...regret]) {
    const regex = new RegExp(`\\b${pattern}\\b`, 'gi');
